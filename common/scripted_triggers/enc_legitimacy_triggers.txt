has_more_legitimacy = {
	
	custom_trigger_tooltip = {
		tooltip = has_legitimacy_tt		
		check_variable = {
			var = enc_legitimacy
			value > legitimacy_check_var
		}
	}
}

has_free_press = {
	OR = {
		has_idea = ENC_free_press
		has_idea = ENC_free_press_2
	}
}

has_no_free_press = {
	not = {
		OR = {
			has_idea = ENC_free_press
			has_idea = ENC_free_press_2
		}
	}
}
